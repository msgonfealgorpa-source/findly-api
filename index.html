const express = require("express"); const cors = require("cors"); const jwt = require("jsonwebtoken"); const bcrypt = require("bcryptjs"); const app = express(); const JWT_SECRET = "findly_ai_secret_2026"; app.use(cors()); app.use(express.json()); // قاعدة بيانات وهمية (تُستبدل بـ MongoDB للإنتاج) const users = []; // دالة حماية المسارات const auth = (req, res, next) => { const token = req.headers.authorization?.split(" ")[1]; if (!token) return res.status(401).json({ message: "غير مصرح" }); try { req.user = jwt.verify(token, JWT_SECRET); next(); } catch { res.status(403).json({ message: "انتهت الجلسة" }); } }; // 1. تسجيل مستخدم جديد (مع الاسم والدولة) app.post("/register", async (req, res) => { const { email, password, name, country, interests } = req.body; if (users.find(u => u.email === email)) return res.status(400).json({ message: "الإيميل مسجل مسبقاً" }); const hashedPassword = await bcrypt.hash(password, 10); const newUser = { id: Date.now(), email, password: hashedPassword, name: name || "مستخدم جديد", country: country || "غير محدد", interests: interests || [] }; users.push(newUser); res.status(201).json({ message: "تم إنشاء الحساب بنجاح" }); }); // 2. تسجيل الدخول app.post("/login", async (req, res) => { const { email, password } = req.body; const user = users.find(u => u.email === email); if (!user || !(await bcrypt.compare(password, user.password))) { return res.status(401).json({ message: "بيانات الدخول غير صحيحة" }); } const token = jwt.sign({ id: user.id }, JWT_SECRET, { expiresIn: '7d' }); res.json({ token }); }); // 3. جلب بيانات الملف الشخصي (كاملة) app.get("/me", auth, (req, res) => { const user = users.find(u => u.id === req.user.id); if (!user) return res.status(404).json({ message: "المستخدم غير موجود" }); const { password, ...userData } = user; // إرسال البيانات بدون الباسورد res.json(userData); }); // 4. محرك البحث (نتائج محاكية بذكاء) app.get("/search", (req, res) => { const q = req.query.q || "منتج"; res.json({ top: [ { name: `${q} - الإصدار الاحترافي`, price: "299$", image: "https://via.placeholder.com/150", link: "#", rating: 5 }, { name: `${q} - الخيار الاقتصادي`, price: "120$", image: "https://via.placeholder.com/150", link: "#", rating: 4 } ] }); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Findly Server running on port ${PORT}`));
